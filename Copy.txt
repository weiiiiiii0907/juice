#include <fcntl.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/stat.h>
#include <sys/types.h>

int main(int argc, char **argv){
    char word[4096];
    int size=0,input,output;
    input=open(argv[1],O_RDONLY);
    output=open(argv[2],O_WRONLY|O_CREAT,00600);
    if(input==-1){
        printf("Open file failed.");
        return 0;
    }
    if(output==-1){
        printf("Open file failed.");
        return 0;
    }
    while(1){
        size=read(input,word,4096);
        if(size<=0) break;
        write(output,word,size);
    }
    close(input);
    close(output);
    return 0;
}
